void QuickSort(vector<int>& array, int first, int last){
    if (first >= last)
        return;
    
    int l = first, r = last;
    int key = array[l];
    while (l < r){
        while (l < r && array[r] >= key){
            r--;
        }
        array[l] = array[r];
        while (l < r && array[l] <= key){
            l++;
        }
        array[r] = array[l];
    }
    array[l] = key;
    
    QuickSort(array, first, l-1);
    QuickSort(array, l+1, last);
}

    while (k < end){
        if (k+1<end && array[k+1]>array[k]){
            k++;
        }
        if (array[start] < array[k]){
            swap(array[start], array[k]);

void ShellSort(vector<int>& array){
    int n = array.size();
    for (int gap=n/2; gap>=1; gap/=2){
        for (int i=0; i<gap; ++i){
            for (int j=i+gap; j<n; j+=gap){
                int tmp = array[j];
                int k = j-gap;
                for (; k>=0 && array[k]>tmp; k-=gap){
                    array[k+gap] = array[k];
                }
                array[k+gap] = tmp;
            }
        }
    }
}

            int tmp = nums[j];
            int k = j-1;
            for (; k>=0 && nums[k]>tmp; --k){
                nums[k+1] = nums[k];
            }
            nums[k+1] = tmp;
        }
    }

    int index = 0;
    for (int i=0; i<nbucket; ++i){
        for (int j=0; j<bucketlist[i].size(); ++j){
            array[index] = bucketlist[i][j];
            index++;
        }
    }
    
}

int getbitnum(vector<int>& array){
    int maxnum = array[0];
    for (int num : array){
        if (num > maxnum)
            maxnum = num;
    }
    int nbit = 0;
    while (maxnum > 0){
        nbit++;
        maxnum /= 10;
    }
    return nbit;
}

void RedixSort(vector<int>& array){

}
